  const fmtDate = d => d ? new Date(d).toISOString().slice(0,10) : '';

  const fmtDateTime = d => {

    if(!d) return ''; const x = new Date(d);

    const pad = n => String(n).padStart(2,'0');

    return `${x.getFullYear()}-${pad(x.getMonth()+1)}-${pad(x.getDate())} ${pad(x.getHours())}:${pad(x.getMinutes())}`;

  }

  const todayKey = () => fmtDate(new Date());

  const uid = p => `${p||'ID'}_${Date.now()}_${Math.random().toString(36).slice(2,8)}`;

  const toNum = v => (v===''||v==null||isNaN(Number(v))) ? 0 : Number(v);

  const parseDate = v => v ? new Date(v) : null;


  // ==========

  // Storage (localStorage)

  // ==========

  const store = {

    key: 'kl_local_hub_v1',

    get(){

      const raw = localStorage.getItem(this.key);

      if(!raw) return { cases:[], touches:[], promises:[], config:{ orgName:'', replyTo:'', owner:'' } };

      try { return JSON.parse(raw) } catch(e){ return { cases:[], touches:[], promises:[], config:{ orgName:'', replyTo:'', owner:'' } } }

    },

    set(data){ localStorage.setItem(this.key, JSON.stringify(data)); },

    wipe(){ localStorage.removeItem(this.key); }

  }


  let DB = store.get();


  // ==========

  // Data helpers

  // ==========

  function addCase(c){ c.id = c.id||uid('CASE'); c.createdAt = c.createdAt||new Date(); c.updatedAt = new Date(); DB.cases.push(c); save(); return c; }

  function updateCase(id, patch){ const c = DB.cases.find(x=>x.id===id); if(!c) return; Object.assign(c, patch, {updatedAt:new Date()}); save(); return c; }

  function addTouch(t){ t.id = t.id||uid('TOUCH'); t.createdAt = t.createdAt||new Date(); DB.touches.push(t); save(); return t; }

  function addPromise(p){ p.id = p.id||uid('PTP'); p.openedAt = p.openedAt||new Date(); DB.promises.push(p); save(); return p; }

  function updatePromise(id, patch){ const p = DB.promises.find(x=>x.id===id); if(!p) return; Object.assign(p, patch); save(); return p; }

  function caseById(id){ return DB.cases.find(x=>x.id===id); }

  function touchesForCase(id){ return DB.touches.filter(t=>t.caseId===id).sort((a,b)=> new Date(b.createdAt)-new Date(a.createdAt)); }

  function promisesForCase(id){ return DB.promises.filter(p=>p.caseId===id).sort((a,b)=> new Date(b.promisedDate)-new Date(a.promisedDate)); }

  function save(){ store.set(DB); renderActive(); }

  // ==========

  // Queue logic

  // ==========

  function computeQueue(){

    const today = new Date(todayKey());


    const ptpDue = DB.promises.filter(p=> p.status==='Open' && p.promisedDate && new Date(p.promisedDate) <= today)

      .map(p=> ({ type:'PTP due', caseId:p.caseId, label:`Promise due ${p.promisedDate}`, weight:100, promised:p }));


    const touchesDue = DB.touches.filter(t=> t.nextActionDate && new Date(t.nextActionDate) <= today)

      .map(t=> ({ type:'Follow-up', caseId:t.caseId, label:`Next action ${t.nextActionDate}`, weight:80, touch:t }));


    const caseNext = DB.cases.filter(c=> c.status==='Open' && c.nextAction && new Date(c.nextAction) <= today && (!c.holdUntil || new Date(c.holdUntil) < today))

      .map(c=> ({ type:'Case action', caseId:c.id, label:`Next action ${c.nextAction}`, weight:60 }));


    const highValue = DB.cases.filter(c=> c.status==='Open' && toNum(c.amount) >= 2000 && (!c.holdUntil || new Date(c.holdUntil) < today))

      .map(c=> ({ type:'High £', caseId:c.id, label:`£${toNum(c.amount).toLocaleString('en-GB')}`, weight:40 }));


    const all = [...ptpDue, ...touchesDue, ...caseNext, ...highValue];

    all.sort((a,b)=> b.weight - a.weight);

    return all;

  }

  function exportCSV(kind){

    let rows=[], header=[];

    if(kind==='cases'){

      header = ['id','customerName','customerEmail','customerPhone','invoiceNumbers','amount','priority','status','nextAction','holdUntil','issueType','owner','createdAt','updatedAt','driveFolder'];

      rows = DB.cases.map(c=> header.map(h=> c[h]??''));

    } else if(kind==='touches'){

      header = ['id','caseId','when','channel','outcome','summary','nextActionDate','createdAt'];

      rows = DB.touches.map(t=> header.map(h=> t[h]??''));

    } else if(kind==='promises'){

      header = ['id','caseId','openedAt','promisedDate','amount','status','followUpDate','notes'];

      rows = DB.promises.map(p=> header.map(h=> p[h]??''));

    }

    const csv = [header.join(','), ...rows.map(r=> r.map(v=> `"${String(v).replace(/"/g,'""')}"`).join(','))].join('\n');

    download(`kindledger_${kind}.csv`, csv, 'text/csv');

  }

  function download(name, data, type){

    const blob = new Blob([data], {type});

    const url = URL.createObjectURL(blob);

    const a = document.createElement('a'); a.href=url; a.download=name; a.click(); URL.revokeObjectURL(url);

  }

  async function importFile(){

    const f = $('#fileImport').files[0]; if(!f) return alert('Pick a file');

    const text = await f.text();

    if(f.type.includes('json')){

      try{ DB = JSON.parse(text); save(); alert('Imported JSON (replaced dataset).'); }catch(e){ alert('Bad JSON'); }

    } else {

      alert('CSV import not implemented in this demo. Export JSON from another machine and import here to migrate.');

    }

  }
